# @one-for-all-ui/react

> React Design Library

## Install

```bash
npm install --save @one-for-all-ui/react
```

## Usage

```tsx
import React, { Component } from 'react'

import { MyComponent } from '@one-for-all-ui/react'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
